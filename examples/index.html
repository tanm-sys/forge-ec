ECDSA Signatures

Beginner secp256k1 ECDSA

Learn how to create and verify ECDSA signatures using the secp256k1 curve, the same cryptography used in Bitcoin.

ecdsa_example.rs 
use forge_ec_core::{Curve, SignatureScheme};
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_signature::ecdsa::Ecdsa;
use forge_ec_rng::os_rng::OsRng;
use sha2::Sha256;

// Generate key pair
let mut rng = OsRng::new();
let secret_key = Secp256k1::random_scalar(&mut rng);
let public_key = Secp256k1::multiply(
    &Secp256k1::generator(),
    &secret_key
);

// Sign message
let message = b"Hello, ECDSA!";
let signature = Ecdsa::::sign(
    &secret_key,
    message
);

// Verify signature
let valid = Ecdsa::::verify(
    &public_key_affine,
    message,
    &signature
);
assert!(valid);

EdDSA Signatures

Beginner Ed25519 EdDSA

Implement Ed25519 signatures for fast and secure digital signatures with built-in protection against many attack vectors.

eddsa_example.rs 
use forge_ec_signature::eddsa::Ed25519Signature;

// Generate private key (32 bytes)
let private_key = [0u8; 32]; // Use secure random bytes
let public_key = Ed25519Signature::derive_public_key(&private_key);

// Sign message
let message = b"Hello, Ed25519!";
let signature = Ed25519Signature::sign(&private_key, message);

// Verify signature
let valid = Ed25519Signature::verify(
    &public_key,
    message,
    &signature
);
assert!(valid);

println!("Ed25519 signature verified: {}", valid);

ECDH Key Exchange

Intermediate secp256k1 ECDH

Perform secure key exchange between two parties using Elliptic Curve Diffie-Hellman to establish a shared secret.

ecdh_example.rs 
use forge_ec_core::Curve;
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_rng::os_rng::OsRng;
use sha2::{Digest, Sha256};

// Alice generates her key pair
let mut rng = OsRng::new();
let alice_sk = Secp256k1::random_scalar(&mut rng);
let alice_pk = Secp256k1::multiply(&Secp256k1::generator(), &alice_sk);

// Bob generates his key pair
let bob_sk = Secp256k1::random_scalar(&mut rng);
let bob_pk = Secp256k1::multiply(&Secp256k1::generator(), &bob_sk);

// Both compute the same shared secret
let alice_shared = Secp256k1::multiply(&bob_pk, &alice_sk);
let bob_shared = Secp256k1::multiply(&alice_pk, &bob_sk);

// Derive symmetric key
let mut hasher = Sha256::new();
hasher.update(&alice_shared_point_affine.x().to_bytes());
let symmetric_key = hasher.finalize();

Schnorr Signatures

Advanced secp256k1 Schnorr

Implement Schnorr signatures with batch verification for improved performance and privacy features.

schnorr_example.rs 
use forge_ec_signature::schnorr::{Schnorr, SchnorrSignature};
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_rng::os_rng::OsRng;

// Generate key pair
let mut rng = OsRng::new();
let secret_key = Secp256k1::random_scalar(&mut rng);
let public_key = Secp256k1::multiply(&Secp256k1::generator(), &secret_key);

// Sign message
let message = b"Hello, Schnorr!";
let signature = Schnorr::::sign(&secret_key, message);

// Verify signature
let valid = Schnorr::::verify(
    &public_key_affine,
    message,
    &signature
);

// Batch verification (multiple signatures)
let signatures = vec![signature];
let public_keys = vec![public_key_affine];
let messages = vec![message];

let batch_valid = SchnorrSignature::batch_verify(
    &public_keys,
    &messages,
    &signatures
);

Key Generation & Encoding

Beginner DER PEM

Generate cryptographic keys and export them in various formats for interoperability with other systems.

keygen_example.rs 
use forge_ec_core::Curve;
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_encoding::{der::EcPrivateKey, pem::PemEncoding};
use forge_ec_rng::os_rng::OsRng;

// Generate key pair
let mut rng = OsRng::new();
let secret_key = Secp256k1::random_scalar(&mut rng);
let public_key = Secp256k1::multiply(&Secp256k1::generator(), &secret_key);

// Export private key in DER format
let private_key_der = EcPrivateKey::new(
    &secret_key.to_bytes(),
    Some(secp256k1_oid()),
    Some(&public_key_bytes)
).to_der()?;

// Export in PEM format
let private_key_pem = PemEncoding::encode(
    "EC PRIVATE KEY",
    &private_key_der
);

println!("Private key (PEM):\n{}", private_key_pem);

OpenSSL Interoperability

Advanced OpenSSL Interop

Create keys and signatures that are compatible with OpenSSL and other cryptographic libraries.

openssl_interop.rs 
use forge_ec_core::{Curve, SignatureScheme};
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_signature::ecdsa::Ecdsa;
use forge_ec_encoding::der::EcdsaSignature;
use sha2::Sha256;

// Create signature compatible with OpenSSL
let secret_key = Secp256k1::random_scalar(&mut rng);
let message = b"OpenSSL compatible message";
let signature = Ecdsa::::sign(&secret_key, message);

// Convert to DER format (OpenSSL standard)
let der_signature = EcdsaSignature::from_signature::(&signature)
    .to_der()?;

// This DER signature can be verified by OpenSSL:
// openssl dgst -sha256 -verify pubkey.pem -signature sig.der message.txt

println!("DER signature: {:?}", hex::encode(&der_signature));
