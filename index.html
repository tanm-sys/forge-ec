ecdsa_example.rs 
use forge_ec_core::{Curve, SignatureScheme};
use forge_ec_curves::secp256k1::Secp256k1;
use forge_ec_signature::ecdsa::Ecdsa;
use forge_ec_rng::os_rng::OsRng;
use sha2::Sha256;

// Generate a key pair
let mut rng = OsRng::new();
let secret_key = Secp256k1::random_scalar(&mut rng);
let public_key = Secp256k1::multiply(
    &Secp256k1::generator(),
    &secret_key
);

// Sign a message
let message = b"Hello, Cryptography!";
let signature = Ecdsa::::sign(
    &secret_key,
    message
);

// Verify the signature
let valid = Ecdsa::::verify(
    &public_key_affine,
    message,
    &signature
);
assert!(valid);

ECDSA Example
Rust secp256k1 ⚠️ Secure 
// Generate key pair and sign message
let mut rng = OsRng::new();
let secret_key = Secp256k1::random_scalar(&mut rng);
let public_key = Secp256k1::multiply(&Secp256k1::generator(), &secret_key);

// Sign and verify
let message = b"Hello, World!";
let signature = Ecdsa::::sign(&secret_key, message);
let valid = Ecdsa::::verify(&public_key_affine, message, &signature);
EdDSA Example
Rust Ed25519 ✓ Fast 
use forge_ec_curves::ed25519::Ed25519;
use forge_ec_signature::eddsa::EdDSA;
use forge_ec_rng::os_rng::OsRng;

// Generate Ed25519 key pair
let mut rng = OsRng::new();
let private_key = Ed25519::random_scalar(&mut rng);
let public_key = Ed25519::derive_public_key(&private_key);

// Sign and verify message
let message = b"Hello, Ed25519!";
let signature = EdDSA::::sign(&private_key, message);
let valid = EdDSA::::verify(&public_key, message, &signature);
ECDH Example
Rust Curve25519 ✓ Secure 
use forge_ec_curves::curve25519::Curve25519;
use forge_ec_ecdh::ECDH;
use forge_ec_rng::os_rng::OsRng;

// Generate key pairs for Alice and Bob
let mut rng = OsRng::new();
let alice_private = Curve25519::random_scalar(&mut rng);
let alice_public = Curve25519::derive_public_key(&alice_private);

let bob_private = Curve25519::random_scalar(&mut rng);
let bob_public = Curve25519::derive_public_key(&bob_private);

// Compute shared secret (same for both parties)
let shared_secret = ECDH::::compute_shared_secret(&alice_private, &bob_public);
